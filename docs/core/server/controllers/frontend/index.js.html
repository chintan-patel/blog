Sets the response context around a post and renders it with the current theme's post view. Used by post preview and single post methods. Returns a function that takes the post to be rendered.

 
function renderPost(req, res) {
    return function renderPost(post) {
        var view = templates.single(req.app.get('activeTheme'), post),
            response = formatResponse.single(post);

        setResponseContext(req, res, response);
        res.render(view, response);
    };
}
